server {

        root /var/www/golf-pool;

        # Add index.php to the list if you are using PHP
        index index.html index.htm index.nginx-debian.html;

        server_name the-golf-pool.com;

        location / {
                try_files $uri /index.html;
        }

        location /api {
        # Proxy requests to your Express server running on localhost:5000
        proxy_pass http://localhost:5000;

        # These are necessary for WebSocket support, which might be useful for future development
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }



    listen [::]:443 ssl ipv6only=on; # managed by Certbot
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/the-golf-pool.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/the-golf-pool.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}




server {
    if ($host = the-golf-pool.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


        listen 80 default_server;
        listen [::]:80 default_server;

	server_name the-golf-pool.com;
	return 404;
}
